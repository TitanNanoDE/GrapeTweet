/**
 * @module Gestures
 * @author Jovan Gerodetti <jovan@titannano.de>
 * @copyright TitnaNano.de 2015
 * @memberof GrapeTweet
 */

$_('grapeTweet').module('Gestures', ['Net', 'Misc'], function(App, ready){

	var { Net, Misc } = App.modules;

	/**
	 * @type {object}
	 * @property {object} metaData
	 * @property {?Node} metaData.rget
	 * @property {?Node} metaData.element
	 * @property {?timeline} metaData.timeline
	 * @property {enabled} metaData.enabled
	 * @property {number} metaData.start
	 * @property {?Touch} metaData.touch
	 * @property {boolean} metaData.open
	 * @property {GrapeTweet.module:Gestures~PullDownToRefresh/release}
	 * @property {GrapeTweet.module:Gestures~PullDownToRefresh/bind}
	 * @property {GrapeTweet.module:Gestures~PullDownToRefresh/listener}
	 * @memberof GrapeTweet.module:Gestures
	 * @inner
	 */
	var PullDownToRefresh = {

		metaData : {
			target : null,
			element : null,
			timeline : null,
			enabled: false,
			start : 0,
			touch : null,
			open : false
		},

		/**
		 * @function PullDownToRefresh/release
		 * @listens touchend
		 * @param {Event} e
		 * @memberof GrapeTweet.module:Gestures
		 * @inner
		 */
		release : function(e){
			if(this.enabled && (!e || e.touches.indexOf(this.touch) < 0)){
				var state= parseInt(this.element.style.getPropertyValue('transform').replace(/translateY\(/, '').replace(/px\)/, ''));
				if(state > 37){
					this.element.classList.add('open');
					this.element.textContent= 'checking for new Tweets...';
					this.open= true;
					var data= this;
					Net.fetchNewHomeTweets(this.timeline).then(function(){
						this.ui.renderTweets(data.timeline).then(function(){
							data.element.classList.add('closed');
							data.element.classList.remove('open');
							data.element.textContent= "release to check for new Tweets";
							data.open= false;
						});
					}, function(){
						data.element.textContent= 'ratelimit exceed, try again later.';
						$$.setTimeout(function(){
							data.element.classList.add('closed');
							data.element.classList.remove('open');
							data.element.textContent= "release to check for new Tweets";
							data.open= false;
						}, 10000);
					});
				}else{
					this.element.classList.add('closed');
				}
				this.start= 0;
				this.enabled= false;
				this.element.style.setProperty('transform', '');
			}
		},

		/**
		 * Binds this gesture to the given element
		 *
		 * @function PullDownToRefresh/bind
		 * @param {Node} element
		 * @param {Timeline} timeline
		 * @param {GrapeTweet.module:UI} UI
		 * @returns	{null}
		 * @memberof GrapeTweet.module:Gestures
		 * @inner
		 */
		bind : function(element, timeline, UI) {
			var metaData = Misc.objectApply(Object.create(PullDownToRefresh.metaData), {
				target : element,
				element : element.querySelector('.check-new'),
				timeline : timeline,
				ui : UI
			});

			return element.addEventListener('touchmove', Misc.throttle(PullDownToRefresh.listener, 100).bind(metaData));
		},

		/**
		 * Main listener for the move event
		 *
		 * @function PullDownToRefresh/listener
		 * @param {Event} e
		 * @listens touchmove
		 * @memberof GrapeTweet.module:Gestures
		 * @inner
		 */
		listener : function (e){
			if((this.enabled || this.target.scrollTop === 0) && !this.open){
				if(!this.enabled){
					this.touch= e.touches.indexOf(e.changedTouches[0]);
					this.start= e.touches[this.touch].screenY;
					this.enabled= true;
					this.target.addEventListener('touchend', this.release.bind(this));
					this.element.classList.remove('closed');
				}else{
					e.preventDefault();
					var force= e.touches[this.touch].screenY - this.start;
					var y= (force / 5);
					if(y < 76){
						this.element.style.setProperty('transform', 'translateY(' + y + 'px)');
					}else{
						PullDownToRefresh.release.apply(this);
					}
				}
			}
		}

	};

	ready({
		PullDownToRefresh : PullDownToRefresh.bind
	});

});
