/**
 * @module Audio
 * @author Jovan Gerodetti
 * @copyright TitnaNano
 * @memberof GrapeTweet
 */

$_('grapeTweet').module('Audio', [], function(App, done){

    /**
     * @property {Audio} recived
     * @property {Audio} sent
     * @property {Audio} snap
     * @memberof GrapeTweet.module:Audio
     * @inner
     */
    var sounds= {
        recieved : new $$.Audio('/sounds/recived.mp3'),
        sent : new $$.Audio('/sounds/sent.mp3'),
        snap : new $$.Audio('/sounds/snap.mp3')
    };

    var Interface = {
        /**
         * plays the given sound
         *
         * @param {string} sound
         * @memberof GrapeTweet.module:Audio
         */
        play : function(sound){
            sounds[sound].play();
        }
    };

    done(Interface);
});
