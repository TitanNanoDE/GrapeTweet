/**
 * @module Misc
 * @author Jovan Gerodetti <jovan@titannano.de>
 * @copyright TitanNano.de 2015
 * @memberof GrapeTweet
 */

$_('grapeTweet').module('Misc', [], function(App, done){

    done({
        /**
         * sorts a list of {Tweet} or {DirectMessage} by date
         *
         * @param {Tweet|DirectMessage} a
         * @param {Tweet|DirectMessage} b
         * @returns {Tweet|DirectMessage}
         * @memberof GrapeTweet.module:Misc
         */
        sortByDate : function(a, b){
            return (( (new $$.Date(a.created_at)).getTime() > (new $$.Date(b.created_at)).getTime() ) ? 1 : -1);
        },

        /**
         * Cuts off a string after the given amount of characters.
         *
         * @param {string} string
         * @param {number} length
         * @returns {string}
         * @memberof GrapeTweet.module:Misc
         */
        cutdown : function(string, length){
            if(string.length > length){
                var new_string= [];
                string= string.split(' ');
                if(string.length == 1){
                    return string[0].substr(0, length) + '...';
                }else{
                    string.forEach(function(item){
                        if( (new_string.join(' ')+' '+item).length <= length )
                            new_string.push(item);
                    });
                    return new_string.join(' ')+'...';
                }
            }else{
                return string;
            }
        },

        /**
         * Splits a string into parts of the given length
         *
         * @param {string} string
         * @param {number} partLength
         * @returns {string[]}
         * @memberof GrapeTweet.module:Misc
         */
        split : function(string, partLength){
            var parts= [];
            var currentPart= [];
            if(string.length > partLength){
                string= string.split(' ');
            }

            while(string.length > 0){
                if((currentPart.join(' ')+ ' ' +string[0]).length <= partLength)
                    currentPart.push(string.shift());
                else{
                    parts.push(currentPart.join(' '));
                    currentPart= [];

                    if(string[0].length > partLength){
                        parts.push(string[0].substr(0, partLength));
                        string[0]= string[0].substr(partLength);
                    }else{
                        currentPart.push(string.shift());
                    }
                }
            }
            parts.push(currentPart.join(' '));

            return parts;
        },

        /**
         * Returns the avatar for the given url with the given type
         *
         * @param {string} type
         * @param {URL} type
         * @returns {URL}
         * @memberof GrapeTweet.module:Misc
         */
        getAvatar : function(type, source) {
            $$.console.log(source);
            var path = source.split('/');
            var fileName = path.pop();
            var extension;

            path = path.join('/');

            extension = fileName.split('_').pop().split('.')[1];
            fileName = fileName.split('_');
            fileName.pop();
            fileName = fileName.join('_');

            return path + '/' + fileName + ((type && type !== 'original') ? '_' + type : '') + '.' + extension;
        },

        /**
         * Applies the given update object on to the given target
         *
         * @param {object} self
         * @param {object} update
         * @param {boolean} noDeepCopy
         * @returns {object}
         * @memberof GrapeTweet.module:Misc
         */
        objectApply : function(self, update, noDeepCopy) {
            update= update || {};

            $$.Object.keys(update).forEach(function(item){
                if (!noDeepCopy && typeof update[item] == 'object' && !$$.Array.isArray(update[item]) && update[item] !== null) {
                    self[item]= self[item] || {};
                    App.objectReplace(self[item], update[item]);
                } else {
                    self[item]= update[item];
                }
            });

            return self;
        },

        /**
         * Wrapps an event listener and limits the amount of executions to once per timeout
         *
         * @param {function} func
         * @param {number} timeout
         * @returns {function}
         * @memberof GrapeTweet.module:Misc
         */
        throttle : function (func, timeout) {
            var last = null;
            var timer = null;

            return function() {
                var context = this;
                var now = Date.now();

                if (last && now < last + timeout) {
                    $$.clearTimeout(timer);
                    timer = $$.setTimeout(function(){
                        func.apply(context, arguments);
                        last = now;
                    }, (last + timeout - now));
                } else {
                    last = now;
                    func.apply(context, arguments);
                }
            };

        }

    });

});
