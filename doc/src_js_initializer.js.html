/**
 * @module Initializer
 * @author Jovan Gerodetti <jovan@titannano.de>
 * @copyright TitanNano.de 2015
 * @memberof GrapeTweet
 */

$_('grapeTweet').module('Initializer', [], function(App, done){

    /**
     * @class
     * @memberof GrapeTweet.module:Initializer
     * @inner
     */
    var Catalog= function(){
        this._listeners= [];
    };

    /**
     * @memberof GrapeTweet.module:Initializer~Catalog
     */
    Catalog.checkListeners= function(){
        var object= this;
        this._listeners.forEach(function(item){
            if(item.event == 'available'){
                var ready= 0;
                item.listener.forEach(function(item){
                    if(Object.keys(object).indexOf(item) > -1)
                        ready++;
                });

                if(ready == item.listener.length){
                    item.success();
                }
            }
        });
    };

    /**
     * @memberof GrapeTweet.module:Initializer~Catalog.prototype
     */
    Catalog.prototype.on= function(event, listener){
        var self= this;
        return new $$.Promise(function(success){
            self._listeners.push({ event : event, listener : listener, success : success });
            Catalog.checkListeners.apply(self);
        });
    };

    /**
     * @memberof GrapeTweet.module:Initializer~Catalog.prototype
     */
    Catalog.prototype.add= function(key, value){
        this[key]= value;
        Catalog.checkListeners.apply(this);
    };

    /**
     * @memberof GrapeTweet.module:Initializer
     * @inner
     */
    var jobs= {};

    /**
     * @memberof GrapeTweet.module:Initializer
     * @inner
     */
    var finishedJobs= new Catalog();

    /**
     * @const
     * @memberof GrapeTweet.module:Initializer
     * @inner
     */
    const LOG= true;

    /**
     * @class TailTip
     * @memberof GrapeTweet.module:Initializer
     * @inner
     */

    var Interface= {
        /**
         * Adds a new job to the bootstrapping process
         *
         * @param {string} name
         * @param {function} job
         * @returns {GrapeTweet.module:Initializer~TailTip}
         * @memberof GrapeTweet.module:Initializer
         */
        job : function(name, job){
            if(name in jobs){
                $$.console.error('initializor job "'+name+'" already exists!');
                return {};
            }

            jobs[name]= {
                job : job,
                dependencies : []
            };

            return {
                /**
                 * Defines the dependencies for the last job.
                 * @param {...string} job
                 * @returns {GrapeTweet.module:Initializer}
                 * @memberof GrapeTweet.module:Initializer~TailTip
                 */
                depends : function(){
                    jobs[name].dependencies= $$.Array.prototype.map.apply(arguments, [function(item){ return item; }]);
                    return Interface;
                },

                /**
                 * @function job
                 * @see GrapeTweet.module:Initializer.job
                 * @memberof GrapeTweet.module:Initializer~TailTip
                 */
                job : Interface.job,

                /**
                 * @function init
                 * @see GrapeTweet.module:Initializer
                 * @memberof GrapeTweet.module:Initializer~TailTip
                 */
                init : Interface.init
            };
        },

        /**
         * Starts the bootstrapping process
         *
         * @returns {Promise}
         * @memberof GrapeTweet.module:Initializer
         */
        init : function(){
            if(LOG) $$.console.time('init');
            jobs= Object.keys(jobs).map(function(key){
                var item= jobs[key];

                return new $$.Promise(function(done){
                    var ready= function(){
                        if(LOG) $$.console.timeEnd(key);
                        finishedJobs.add(key, true);
                        done();
                    };

                    if(!item.dependencies.length){
                        if(LOG) $$.console.time(key);
                        item.job(ready);
                    }else{
                        finishedJobs.on('available', item.dependencies).then(function(){
                            if(LOG) $$.console.time(key);
                            item.job(ready);
                        });
                    }
                });
            });

            return $$.Promise.all(jobs).then(function(){
                if(LOG) $$.console.timeEnd('init');
            });
        }
    };

    done(Interface);
});
